67  inline af::array conv2d(const conv_params_t &params,
68  const af::array &input,
69  const af::array &filters,
70  const af::array &bias,
71  const bool &has_bias){
72  long Cin = input.dims(2); long Cout = filters.dims(3); long batch = input.dims(3);
73 
74  if (has_bias)
75  assert(bias.dims(2) == Cout);
76 
77  assert(filters.dims(2) == Cin);
78  assert(filters.dims(0) == params.filter_x);
79  assert(filters.dims(1) == params.filter_y);
80 
81  long h_in = input.dims(0); long w_in = input.dims(0);
82  long h_out = (input.dims(0) - params.filter_x + 2*params.pad_x)/params.stride_x + 1;
83  long w_out = (input.dims(1) - params.filter_y + 2*params.pad_y)/params.stride_y + 1;
84 
85  af::array out = af::constant(0, h_out, w_out, Cout, batch); // (h, w, k, n)
86 
87  af::array in = af::unwrap(input, params.filter_x, params.filter_y,
88  params.stride_x, params.stride_y,
89  params.pad_x, params.pad_y);
90  af::array f = af::reorder(filters, 3, 0, 1, 2);
91  f = af::moddims(f, Cout, params.filter_x*params.filter_y, Cin); // (Cout, h*w, Cin)
92  af::array b;
93  if (has_bias)
94  b = af::tile(bias, h_out, w_out, 1, 1);
95 
96  // input is (fx*fy, ox*oy, Cin, n)
97  // filters is (Cout, fx*fy, Cin)
98  // need to multiply each of Cout filters onto input
99  // so take Cout x fx*fy . fx*fy x ox*oy = Cout x ox*oy
100 
101  // Once batched matmul is available, gotta get rid of these for loops (or just replace with gfor?)
102 #pragma omp parallel for // parallelize the batch dimension, maybe we can replace in the future?
103  for (int i = 0; i < batch; i++){
104  for (int k = 0; k < Cin; k++){
105  out(af::span, af::span, af::span, i) +=
106  af::moddims(
107  af::reorder(
108  af::matmul(f(af::span, af::span, k, af::span), in(af::span, af::span, k, i)),
109  3, 1, 2, 0),
110  h_out, w_out, Cout);
111  }
112  if (has_bias)
113  out(af::span, af::span, af::span, i) += b;
114  }
115 
116  return out;
117 
118  }

128  inline af::array linear(const af::array &input,
129  const af::array &weight,
130  const af::array &bias,
131  const bool &has_bias){
132  long batch = input.dims(3);
133  long flat = input.dims(0) * input.dims(1) * input.dims(2);
134 
135  assert(weight.dims(1) == flat);
136 
137  if (has_bias)
138  assert(bias.dims(0) == weight.dims(0));
139 
140  af::array out;
141  if (has_bias)
142  out = af::tile(bias, 1, batch) + af::matmul(weight, af::moddims(af::transpose(input), flat, batch));
143  else
144  out = af::matmul(weight, af::moddims(af::transpose(input), flat, batch));
145 
146  out = af::reorder(out, 0, 3, 2, 1);
147 
148  return out;
149 
150  }

161  inline std::vector<af::array> split_branch(const af::array &input, const int &slices, const int &dim){
162  assert(input.dims(dim) % slices == 0); // it's gotta be an even split
163  assert(slices <= 4); // we can only concatenate 4 items so don't split more than that
164  std::vector<af::array> out;
165  switch (dim) { // each array has a slice that is an equal piece of that dimension. Also af::seq is inclusive.
166  case 0 : // change to gfor
167  for (int i = 0; i < input.dims(0); i += input.dims(0)/slices) {
168  out.push_back(input(af::seq(i, i+input.dims(0)/slices-1), af::span, af::span, af::span));
169  }
170  break;
171 
172  case 1 :
173  for (int i = 0; i < input.dims(1); i += input.dims(1)/slices) {
174  out.push_back(input(af::span, af::seq(i, i+input.dims(1)/slices-1), af::span, af::span));
175  }
176  break;
177 
178  case 2 :
179  for (int i = 0; i < input.dims(2); i += input.dims(2)/slices) {
180  out.push_back(input(af::span, af::span, af::seq(i, i+input.dims(2)/slices-1), af::span));
181  }
182  break;
183 
184  case 3 :
185  for (int i = 0; i < input.dims(3); i += input.dims(3)/slices) {
186  out.push_back(input(af::span, af::span, af::span, af::seq(i, i+input.dims(3)/slices-1)));
187  }
188  break;
189 
190  default: out.push_back(input); break; //no op if dim isn't in range.
191  }
192  return out;
193  }

217  inline af::array batchnorm2d(const af::array &gamma, // vectors of size C (applied across channels)
218  const af::array &beta,
219  const af::array &running_mean,
220  const af::array &running_variance,
221  const float &epsilon,
222  const af::array &input){
223 
224  assert(gamma.dims(2) == input.dims(2));
225  assert(beta.dims(2) == input.dims(2));
226  int h_in = input.dims(0); int w_in = input.dims(1); int batch = input.dims(3);
227 
228  af::array out = (input - af::tile(running_mean, h_in, w_in, 1, batch))
229  /(af::sqrt(af::tile(running_variance, h_in, w_in, 1, batch)) + epsilon)
230  * af::tile(gamma, h_in, w_in, 1, batch) + af::tile(beta, h_in, w_in, 1, batch);
231 
232  return out;
233  }

235  inline af::array maxpool(const pooling_params_t &params,
236  const af::array &input,
237  af::array &indices){
238 
239  int h_in = input.dims(0); int w_in = input.dims(1);
240  int h_out = (int)floor((h_in + 2*params.pad_x - (params.filter_x - 1) - 1)/params.stride_x + 1);
241  int w_out = (int)floor((w_in + 2*params.pad_y - (params.filter_y - 1) - 1)/params.stride_y + 1);
242 
243  af::array in = af::unwrap(input, params.filter_x, params.filter_y,
244  params.stride_x, params.stride_y,
245  params.pad_x, params.pad_y);
246 
247  af::array maxima, idx;
248  af::max(maxima, idx, in, 0);
249  af::array out = af::array(maxima, h_out, w_out, input.dims(2), input.dims(3));
250  indices = af::array(idx, h_out, w_out, input.dims(2), input.dims(3));
251 
252  return out;
253
