72  inline af::array forward(const af::array &input){
73  std::vector<af::array> out; // first input is input
74  out.push_back(input);
75  for (int i = 0; i < layers.size(); i++){
76  std::vector<af::array> temp;
77  if (layers[i].size() == 1) { // works for concat layers
78  assert(out.size() <= 4);
79  // Call forward function
80  temp = layers[i][0]->forward(out);
81  }
82  else{
83  temp.clear();
84  assert(out.size() == layers[i].size()); // make sure there are enough inputs
85  for (int j = 0; j < layers[i].size(); j++){
86  // Call forward for each branch
87  temp.push_back(layers[i][j]->forward({out[j]})[0]);
88  }
89  }
90  out = temp;
91  }
92 
93  return out[0]; // must be a single tensor by the end
94  }
